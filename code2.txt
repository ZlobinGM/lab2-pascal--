program Example;
var
    num1, num2: integer;
begin
    num1 := 5;
    Write("Введите отличное от нуля целое число: ");
    Read(num2);
    if (num2 = 0) then
        Write("Неверный ввод");
    else
    begin
        num1 := (num1 + num2 * 5 div 2 - 11) / ((num1 div 2) * 2);
        Write("Целая часть результата  = ", num1);
    end
end.
